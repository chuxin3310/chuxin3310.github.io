## code：

```
#include<bits/stdc++.h>
using namespace std;
int n,m,s,nmax=-1,k=0;
int main()
{
	cin >> n >> m;
	int num[n][m];
	int maxi[n];
	for(int a=0;a<n;a++){
		for(int b=0;b<m;b++){
			cin >> num[a][b];
		}
	}
	for(int a=0;a<n;a++){
		for(int b=0;b<m;b++){
			if(num[a][b]>nmax){
				nmax = num[a][b];
			}
		}
		maxi[a]=nmax;
		nmax = -1;
	}
	for(int a=0;a<n;a++){
		s += maxi[a];
	}
	cout << s << endl;
	for(int a=0;a<n;a++){
		if(s%maxi[a]==0 && k==0){
			cout << maxi[a];
			k=1;
		}
		else if(s%maxi[a]==0 && k!=0){
			cout << " " << maxi[a];
			k=1;
		}
	}
	if(k==0){
		cout << "-1" ;
	}
	return 0;
}
```
