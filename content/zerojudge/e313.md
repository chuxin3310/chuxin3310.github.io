#include<bits/stdc++.h>
using namespace std;
vector <string> str;
char ch;
string s;
int c[26],amoun,op=0;
int mindfs = 27;
int main(){
    cin >> amoun;
    int dfs[amoun];
	memset(dfs,0,sizeof(dfs));
    string estr[amoun];
    for(int a=0;a<amoun;a++){
        cin >> s;
        str.push_back(s); //把所有string存入vector中

        for(int b=0;b<s.size();b++){
            ch = s[b];
            if(c[int(ch)-65] == 0){
                c[int(ch)-65] = 1;
                dfs[a]++;//判斷每個string共有幾個相異字元
            }            
        }

        if(dfs[a]<mindfs){
            mindfs = dfs[a];
        }  

        memset(c,0,sizeof(c)); 
              
     }
     for(int a=0;a<amoun;a++){
        if(dfs[a]== mindfs){
            estr[op]=str[a];
            op++;
        }
    } 
	 sort(estr,estr+op);
        cout << estr[0];
}
