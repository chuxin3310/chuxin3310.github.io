#include<bits/stdc++.h>
using namespace std;
int it=-1,n=0; 
string s;
int pix(int n){
    it++;
    if(s[it]=='2'){
        return pix(n/2)+pix(n/2)+pix(n/2)+pix(n/2);
    }
    if(s[it]=='1'){
        return n*n;
    }
    else{
        return 0;
    }
}
int main(){
    cin >> s >> n;
    cout << pix(n);
}
